490static int crypto_tr_lift_x(const uint8_t x[static 32], uint8_t out[static 65]) {
491 // save memory by reusing output buffer for intermediate results
492 uint8_t *y = out + 1 + 32;
493 // we use the memory for the x-coordinate of the output as a temporary variable
494 uint8_t *c = out + 1;
495
496 uint8_t e = 3;
497 cx_math_powm(c, x, &e, 1, secp256k1_p, 32); // c = x^3 (mod p)
498 uint8_t scalar[32] = {0};
499 scalar[31] = 7;
500 cx_math_addm(c, c, scalar, secp256k1_p, 32); // c = x^3 + 7 (mod p)
501
502 cx_math_powm(y, c, secp256k1_sqr_exponent, 32, secp256k1_p, 32); // y = sqrt(x^3 + 7) (mod p)
503
504 // sanity check: fail if y * y % p != x^3 + 7
505 uint8_t y_2[32];
506 e = 2;
507 cx_math_powm(y_2, y, &e, 1, secp256k1_p, 32); // y^2 (mod p)
508 if (cx_math_cmp(y_2, c, 32) != 0) {
509 return -1;
510 }
511
512 if (y[31] & 1) {
513 // y must be even: take the negation
514 cx_math_sub(out + 1 + 32, secp256k1_p, y, 32);
515 }
516
517 // add the 0x04 prefix; copy x verbatim
518 out[0] = 0x04;
519 memcpy(out + 1, x, 32);
520
521 return 0;
522}
